function doPost(e){
  const data = JSON.parse(e.postData.contents);
  const sheet = SpreadsheetApp.openById("YOUR_SHEET_ID").getSheetByName("Pledges");
  sheet.appendRow([data.id, data.name, data.state, data.profile, data.hearts, data.date]);
  return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON);
}
